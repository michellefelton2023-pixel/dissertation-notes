<script>
    let notes = JSON.parse(localStorage.getItem('diss_notes')) || [];

    function saveNote() {
        const author = document.getElementById('author').value;
        if(!author) return alert("Please enter an author.");

        const note = {
            id: Date.now(),
            author,
            title: document.getElementById('title').value,
            quote: document.getElementById('quote').value,
            meaning: document.getElementById('meaning').value,
            seed: document.getElementById('seed').value,
            lenses: Array.from(document.querySelectorAll('.lens:checked')).map(el => el.value),
            date: new Date().toLocaleDateString()
        };

        notes.unshift(note);
        localStorage.setItem('diss_notes', JSON.stringify(notes));
        renderNotes();
        clearForm();
    }

    function renderNotes() {
        const container = document.getElementById('notesContainer');
        document.getElementById('count').innerText = notes.length;
        container.innerHTML = notes.map(n => `
            <div class="note-card">
                <div class="note-source">${n.author} ${n.title ? ' — ' + n.title : ''}</div>
                ${n.quote ? `<div class="note-quote">"${n.quote}"</div>` : ''}
                <div style="font-size:0.8rem; margin-top:8px"><strong>Analysis:</strong> ${n.meaning}</div>
                <div style="margin-top:10px">
                    ${n.lenses.map(l => `<span class="note-lens-pill">${l}</span>`).join('')}
                </div>
                <div style="font-size:0.6rem; color:var(--muted); margin-top:10px">${n.date}</div>
            </div>
        `).join('');
    }

    function clearForm() {
        ['author','title','quote','meaning','seed'].forEach(id => document.getElementById(id).value = '');
        document.querySelectorAll('.lens').forEach(el => el.checked = false);
    }

    function openExport() {
        document.getElementById('modal').classList.add('open');
        document.getElementById('exportContent').value = JSON.stringify(notes, null, 2);
    }

    function copyData() {
        document.getElementById('exportContent').select();
        document.execCommand('copy');
        alert("Data copied!");
    }

    function clearData() {
        if(confirm("Permanently delete all notes?")) {
            notes = [];
            localStorage.removeItem('diss_notes');
            renderNotes();
            document.getElementById('modal').classList.remove('open');
        }
    }

    renderNotes();
</script>
